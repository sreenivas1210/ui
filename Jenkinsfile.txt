pipeline {
    agent any


    stages {
        stage('Build') {
            steps {
                echo 'Build'
                sh 'npm install'
            }
        }

        stage('second stage') {
            steps {
                echo 'second stage'

                sh "npm run build"
            }
        }

        stage('Deploy') {
            steps {
                echo 'in deploy stage'

                sh "aws lambda update-function-code --function-name $function_name --region us-east-1 --s3-bucket vasuui --s3-key sample-1.0.3.jar"
            }
        }
    }
}
